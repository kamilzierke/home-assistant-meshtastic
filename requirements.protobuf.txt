# Only needed for scripts/generate_protobufs.
# grpcio-tools provides the protoc compiler (invoked as `python -m grpc_tools.protoc`);
# mypy-protobuf provides its --mypy_out plugin for generating *.pyi stubs.
#
# IMPORTANT: grpcio-tools bundles a native protoc binary with a FIXED gencode
# version baked in at build time - it is NOT controlled by whatever `protobuf`
# package happens to be pip-installed alongside it. That gencode version becomes
# the generated code's minimum required `protobuf` runtime version (see the
# comment at the top of scripts/generate_protobufs), and it must not exceed the
# EXACT version Home Assistant core pins in its own
# homeassistant/package_constraints.txt (currently protobuf==6.32.0 for
# homeassistant==2026.7.2 - check
# https://github.com/home-assistant/core/blob/<HA version tag>/homeassistant/package_constraints.txt
# for the version you're targeting), since HA rejects any custom-integration
# requirement that doesn't satisfy its own pin (RequirementsNotFound) - it will
# NOT resolve to something newer to satisfy us.
#
# grpcio-tools==1.76.0 was verified (2026-07) to produce gencode "6.31.1", which
# satisfies HA core 2026.7.2's protobuf==6.32.0 pin. If you bump either pin,
# regenerate and check the "Protobuf Python Version: X.Y.Z" comment in any
# regenerated *_pb2.py file against HA core's current package_constraints.txt
# before committing.
grpcio-tools==1.76.0
mypy-protobuf==3.7.0
